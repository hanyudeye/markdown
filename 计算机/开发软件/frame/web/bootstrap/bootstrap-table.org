*** 用法

**** 通过 data-url 方式
     #+begin_src html
       <table
         data-toggle="table"
         data-url="data1.json"
         data-pagination="true"
         data-search="true">
         <thead>
           <tr>
             <th data-sortable="true" data-field="id">Item ID</th>
             <th data-field="name">Item Name</th>
             <th data-field="price">Item Price</th>
           </tr>
         </thead>
       </table>
     #+end_src

**** 通过 javascript 方式
     
     <table id="table"></table>

     #+begin_src js
       $('#table').bootstrapTable({
           columns: [{
               field: 'id',
               title: 'Item ID'
           }, {
               field: 'name',
               title: 'Item Name'
           }, {
               field: 'price',
               title: 'Item Price'
           }],
           data: [{
               id: 1,
               name: 'Item 1',
               price: '$1'
           }, {
               id: 2,
               name: 'Item 2',
               price: '$2'
           }]
       })
  #+end_src

**** 通过远程
     #+begin_src js
       $('#table').bootstrapTable({

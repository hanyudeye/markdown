js 用法

$('#myTab a').on('click', function (event) { event.preventDefault() $(this).tab('show') }) $('#myTab a[href="#profile"]').tab('show') // Select tab by name $('#myTab li:first-child a').tab('show') // Select first tab $('#myTab li:last-child a').tab('show') // Select last tab $('#myTab li:nth-child(3) a').tab('show') // Select third tab
